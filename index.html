import os import uuid import openai import requests from moviepy.editor import * from googleapiclient.discovery import build from googleapiclient.http import MediaFileUpload from oauth2client.service_account import ServiceAccountCredentials # ================= CONFIG ================= OPENAI_API_KEY = "YOUR_OPENAI_KEY" ELEVEN_API_KEY = "YOUR_ELEVENLABS_KEY" VOICE_ID = "YOUR_VOICE_ID" YOUTUBE_CLIENT_SECRET = "client_secret.json" openai.api_key = OPENAI_API_KEY if not os.path.exists("output"): os.makedirs("output") # ================= STEP 1: SCRIPT ================= def generate_script(topic): res = openai.ChatCompletion.create( model="gpt-4o-mini", messages=[{"role": "user", "content": f"اكتب سكريبت فيديو قصير وجذاب عن: {topic}"}] ) return res["choices"][0]["message"]["content"] # ================= STEP 2: VOICE ================= def generate_voice(text): filename = f"output/{uuid.uuid4()}.mp3" url = f"https://api.elevenlabs.io/v1/text-to-speech/{VOICE_ID}" headers = { "xi-api-key": ELEVEN_API_KEY, "Content-Type": "application/json" } data = {"text": text} res = requests.post(url, json=data, headers=headers) with open(filename, "wb") as f: f.write(res.content) return filename # ================= STEP 3: VIDEO ================= def create_video(text, audio_path): video_path = f"output/{uuid.uuid4()}.mp4" audio = AudioFileClip(audio_path) duration = audio.duration bg = ColorClip((720,1280), color=(0,0,0), duration=duration) txt = TextClip( text[:250], fontsize=50, color='white', size=(700,1000), method='caption' ).set_position('center').set_duration(duration) video = CompositeVideoClip([bg, txt]) video = video.set_audio(audio) video.write_videofile(video_path, fps=24) return video_path # ================= STEP 4: UPLOAD TO YOUTUBE ================= def upload_to_youtube(video_file, title): youtube = build("youtube", "v3", developerKey="YOUR_YOUTUBE_API_KEY") request = youtube.videos().insert( part="snippet,status", body={ "snippet": { "title": title, "description": "AI generated video", "tags": ["AI", "shorts"], "categoryId": "22" }, "status": { "privacyStatus": "public" } }, media_body=MediaFileUpload(video_file) ) response = request.execute() print("Uploaded:", response) # ================= RUN ================= if __name__ == "__main__": topic = input("Enter video topic: ") print("Generating script...") script = generate_script(topic) print("Generating voice...") audio = generate_voice(script) print("Creating video...") video = create_video(script, audio) print("Uploading to YouTube...") upload_to_youtube(video, topic) print("DONE ✅")
